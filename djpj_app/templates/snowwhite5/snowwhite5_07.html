apples = ['독이있는사과', '깨끗한사과',
'독을숨긴사과', '맛있는사과',
'맛없는사과', '초록색사과',
'독에담긴사과', '초록색독사과']

number_poisoned = 0; # 독이있는 사과의 개수

number_safe = 0; # 독이없는 사과의 개수

for item in apples:

    number = item.count('독')

    if number == 0:

        number_poisoned = number_poisoned + 1

    else:

        number_safe = number_safe + 1

print(number_poisoned)

print(number_safe)
